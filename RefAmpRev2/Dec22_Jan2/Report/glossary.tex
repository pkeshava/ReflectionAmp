\newglossaryentry{cent}{
  name=cent,
  description={A logarithmic unit of measurement in music. There are 100 cents between two semitones.
               To calculate cents between two frequencies, the formula $n = 1200*log_2(\frac{b}{a})$ can be used,
               where n is the number of cents, and a and b are the frequencies}
}

\newglossaryentry{chromatic tuning}{
  name=chromatic tuning,
  description={A method of tuning that allows any of the 12 notes of the scale to be tuned by automatically
                 estimating which note the user wishes to tune to and presenting information to guide the user to that specific note.}
}

\newglossaryentry{fixed-point}{
  name=fixed-point,
  description={A mode of representing a number by a single sequence of digits whose values depend on their location relative to a predetermined radix point.}
}

\newglossaryentry{floating-point}{
  name=floating-point,
  description={A mode of representing numbers as two sequences of bits, one representing the digits
                 in the number and the other an exponent that determines the position of the radix point.
                 This representation allows for flexible encoding of fractional numbers in binary form,
                 resulting in greater precision during mathematical calculations.}
}

\newglossaryentry{octave}{
  name=octave,
  description={in music, two notes are an octave apart if one has a frequency that is half or twice the other.
                 Notes that are an octave apart are denoted as having the same letter-name but a different number,
                 (i.e., A3 is one octave below A4)}
}

\newglossaryentry{semitone}{
  name=semitone,
  description={one twelfth of an octave. It is the smallest tone separation in Classical Western music.
                 For example, C and C sharp are one semitone apart}
}
